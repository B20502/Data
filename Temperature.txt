#include <stdio.h>

float celsius_to_fahrenheit(float celsius) {
    return (celsius * 9 / 5) + 32;
}

float fahrenheit_to_celsius(float fahrenheit) {
    return (fahrenheit - 32) * 5 / 9;
}

int main() {
    int choice;
    float temperature, converted_temperature;

    printf("Temperature Converter\n");
    printf("1: Convert from Celsius to Fahrenheit\n");
    printf("2: Convert from Fahrenheit to Celsius\n");
    
    printf("Choose an option (1 or 2): ");
    scanf("%d", &choice);
    
    if (choice == 1) {
        printf("Enter temperature in Celsius: ");
        scanf("%f", &temperature);
        converted_temperature = celsius_to_fahrenheit(temperature);
        printf("%.2f째C is equal to %.2f째F\n", temperature, converted_temperature);
    } else if (choice == 2) {
        printf("Enter temperature in Fahrenheit: ");
        scanf("%f", &temperature);
        converted_temperature = fahrenheit_to_celsius(temperature);
        printf("%.2f째F is equal to %.2f째C\n", temperature, converted_temperature);
    } else {
        printf("Invalid choice. Please select 1 or 2.\n");
    }

    return 0;
}
